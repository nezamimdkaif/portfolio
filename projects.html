uav_sonar_docking.cppC++ / Arduino
Firmware loop for precision ultrasonic sensor reading and autonomous landing alignment check on the ISRO IROC 2026 Docking Station.

3// Sensor Pin Definitions
4const int TRIG_PIN = 9;
5const int ECHO_PIN = 10;
6const int ALIGN_INDICATOR = 13;
7
8// Telemetry thresholds
9const float TARGET_DOCK_DIST_CM = 15.0;
10const float ALIGN_TOLERANCE_CM = 2.0;
11
12void setup() {
13  pinMode(TRIG_PIN, OUTPUT);
14  pinMode(ECHO_PIN, INPUT);
15  pinMode(ALIGN_INDICATOR, OUTPUT);
16  Serial.begin(115200);
17  Serial.println("[SYSTEM]: Sonar Docking Subsystem Initialized.");
18}

20float measureDistance() {
21  digitalWrite(TRIG_PIN, LOW);
22  delayMicroseconds(2);
23  digitalWrite(TRIG_PIN, HIGH);
24  delayMicroseconds(10);
25  digitalWrite(TRIG_PIN, LOW);
26  
27  long duration = pulseIn(ECHO_PIN, HIGH);
28  float distance = duration * 0.034 / 2.0; // cm
29  return distance;
30}
31
32void loop() {
33  float currentDist = measureDistance();
34  
35  // Verify alignment parameters
36  bool isAligned = (abs(currentDist - TARGET_DOCK_DIST_CM) <= ALIGN_TOLERANCE_CM);
37  
38  if (isAligned) {
39    digitalWrite(ALIGN_INDICATOR, HIGH);
40    Serial.println("[TELEMETRY]: UAV Aligned. Sending Docking Trigger.");
41  } else {
42    digitalWrite(ALIGN_INDICATOR, LOW);
43  }
44  
45  delay(100); // 10Hz Telemetry Rate
46}
